// MUX with exact inputs
(MUX s1 (MUX s0 a b) (MUX s0 a b)) // Simplifies to a (LUT 202 s0 a b)

// MUX with match cuts
(MUX s1 (MUX s0 a b) (MUX s0 b a)) // Simplifies to a (LUT 44234 s1 s0 b a)

// 4:1 mux
(MUX s1 (MUX s0 a b) (MUX s0 c d)) // Simplifies to (LUT 51952 s1 (LUT 61642 s1 s0 c d) a b) using DSD

// We can create the circuit by hand too
(NOT (NOR (AND s0 a) (AND (NOT s0) b)))
